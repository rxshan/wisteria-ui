# Transition

```tsx
import { Transition } from 'wisteria-ui';
```

## 用法

### Zoom

```tsx
<Transition.Zoom in>Zoom</Transition.Zoom>
```

<CodeDemoArea label="Zoom Transition">
  <AnimateDemo defaultVisible TransitionComponent={Transition.Zoom} />
</CodeDemoArea>

### Fade

```tsx
<Transition.Fade in>Fade</Transition.Fade>
```

<CodeDemoArea label="Fade Transition">
  <AnimateDemo TransitionComponent={Transition.Fade} />
</CodeDemoArea>

### Collapse

```tsx
<Transition.Collapse in>
  <div>Collapse</div>
</Transition.Collapse>
```

<CodeDemoArea label="Collapse Transition">
  <CollapseDemo />
</CodeDemoArea>
