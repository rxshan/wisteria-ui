# Badge

```tsx
import { Badge } from 'wisteria-ui';
```

## 用法

### 变体类型

```tsx
<Badge variant="dot" />
<Badge variant="standard" />
```

<CodeDemoArea label="Variant">
  <div class="flex gap-8">
    <Badge variant="dot">
      <div class="w-16 h-16 flex items-center justify-center bg-violet-400 text-white rounded-md shadow-sm"></div>
    </Badge>
    <Badge count={9999} max={99}>
      <div class="w-16 h-16 flex items-center justify-center bg-violet-400 text-white rounded-md shadow-sm"></div>
    </Badge>
  </div>
</CodeDemoArea>

### 是否显示

```tsx
<Badge invisible={false} />
```

<Callout type="warning">当 **invisible** 为真时，将会隐藏组件。</Callout>

<CodeDemoArea label="Invisible">
  <BadgeToggleDemo />
</CodeDemoArea>

### 自定义背景色

```tsx
<Badge color="orange" />
<Badge color="green" />
<Badge color="violet" />
```

<CodeDemoArea label="Color">
  <div class="flex gap-12">
    <Badge color="orange" count={9} max={9}>
      <div class="w-16 h-16 flex items-center justify-center bg-violet-400 text-white rounded-md shadow-sm"></div>
    </Badge>
    <Badge color="green" count={99} max={99}>
      <div class="w-16 h-16 flex items-center justify-center bg-violet-400 text-white rounded-md shadow-sm"></div>
    </Badge>
    <Badge color="violet" count={999} max={999}>
      <div class="w-16 h-16 flex items-center justify-center bg-violet-400 text-white rounded-md shadow-sm"></div>
    </Badge>
  </div>
</CodeDemoArea>

### 显示数值

```tsx
<Badge count={0} />
<Badge showZero count={0} />
<Badge count={999} max={999} />
```

<Callout type="info">
  当设置 **showZero** 时，组件将会显示出数字 `0`，否则将会隐藏。
</Callout>

<CodeDemoArea>
  <div class="flex gap-12">
    <Badge count={0}>
      <div class="w-16 h-16 flex items-center justify-center bg-violet-400 text-white rounded-md shadow-sm"></div>
    </Badge>
    <Badge showZero count={0}>
      <div class="w-16 h-16 flex items-center justify-center bg-violet-400 text-white rounded-md shadow-sm"></div>
    </Badge>
    <Badge count={999} max={999}>
      <div class="w-16 h-16 flex items-center justify-center bg-violet-400 text-white rounded-md shadow-sm"></div>
    </Badge>
  </div>
</CodeDemoArea>

### 偏移位置

```tsx
<Badge variant="dot" offset={["-100%", 0]} />
<Badge count={999} offset={[0, "-100%"]} />
<Badge count={999} offset={[120, "0"]} />
```

<Callout type="info">**offset** 偏移位置基于右上角坐标。</Callout>

<CodeDemoArea label="Offset">
  <div class="flex gap-12">
    <Badge variant="dot" offset={['-100%', 0]}>
      <div class="w-16 h-16 flex items-center justify-center bg-violet-400 text-white rounded-md shadow-sm"></div>
    </Badge>
    <Badge count={999} offset={[0, '-100%']}>
      <div class="w-16 h-16 flex items-center justify-center bg-violet-400 text-white rounded-md shadow-sm"></div>
    </Badge>
    <Badge count={999} offset={[20, '0']}>
      <div class="w-16 h-16 flex items-center justify-center bg-violet-400 text-white rounded-md shadow-sm"></div>
    </Badge>
  </div>
</CodeDemoArea>
