# Paper

```tsx
import { Paper } from 'wisteria-ui';
```

## 用法

### Variant

```tsx
<Paper variant="outlined">
  <Box width={180} height={80}>Outlined</Box>
</Paper>

<Paper variant="elevation" elevation={3}>
  <Box width={180} height={80}>Elevation</Box>
</Paper>
```

<CodeDemoArea label="预览">

        height={80}
        className="bg-transparent dark:text-white text-black"
      >
        Outlined
      </Box>
    </Paper>
    <Paper variant="elevation" elevation={3} corner={12}>

</CodeDemoArea>

### Elevation

```tsx
Array.from({ length: 6 }, (_, index) => index).map(elevation => (
  <Paper elevation={elevation}>
    <Box
      width={180}
      height={80}
      className="bg-transparent dark:text-white text-black"
    >
      {`elevation-${elevation}`}
    </Box>
  </Paper>
));
```

<CodeDemoArea label="预览">
  <Space wrap size={24}>
    {Array.from({ length: 6 }, (_, index) => index).map(elevation => (
      <Paper elevation={elevation}>
        <Box
          width={180}
          height={80}
          className="bg-transparent dark:text-white text-black"
        >
          {`elevation-${elevation}`}
        </Box>
      </Paper>
    ))}
  </Space>
</CodeDemoArea>

### Corner

```tsx
<Paper corner="999px" elevation={3}>
  <Box>Corner - Full</Box>
</Paper>
```
