# Radio

```tsx
import { Radio } from 'wisteria-ui';
```

## 用法

### 基本使用

```tsx
<Radio>单选框</Radio>
<Radio disabled>禁用</Radio>
```

<CodeDemoArea label="Usage">
  <Radio>单选框</Radio>
  <Radio disabled>禁用</Radio>
</CodeDemoArea>

### 自定义颜色

```tsx
<Radio checked color="red">红色</Radio>
<Radio checked color="orange">橙色</Radio>
<Radio checked color="violet">紫罗兰色</Radio>
```

<CodeDemoArea label="Color">
  <Radio checked color="red">
    红色
  </Radio>
  <Radio checked color="orange">
    橙色
  </Radio>
  <Radio checked color="violet">
    紫罗兰色
  </Radio>
</CodeDemoArea>

### 单选框组

```tsx
<Radio.Group name="demo" defaultValue="A">
  <Radio value="A">A</Radio>
  <Radio value="B">B</Radio>
  <Radio value="C">C</Radio>
  <Radio value="D">D</Radio>
</Radio.Group>
```

</CodeDemoArea>
